const { Events } = require('discord.js');

module.exports = {
	name: Events.MessageCreate,
	async execute(message) {
        if (message.author.bot) return;
        
        if (message.content == 'タイマーを起動して') { //反応するメッセージ(完全一致)     

            await message.channel.send('');//送信するメッセージ
            
        }
	},
};
